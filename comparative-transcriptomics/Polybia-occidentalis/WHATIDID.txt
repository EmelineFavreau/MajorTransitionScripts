# Aim 1: Run Nextflow

mkdir -p tmp/star-one-sample

mkdir -p /lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/result/star-one-sample

cp myriad.config tmp/star-one-sample/.

cd tmp/star-one-sample

tmux new-session

module load blic-modules

module load nextflow/19.10.0.5170

nextflow run nf-core/rnaseq \
	-profile singularity \
	--reads '/lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/input/PO_Q1_*{1,2}.fq.gz' \
	--fasta '/lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/input/P.occidentalis.RM_bactfilt.min500.euk.fasta.gz' \
	--gff '/lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/input/P.occidentalis.evm.consensus.annotation.v2a.gff3.gz' \
	--forwardStranded \
	--skipBiotypeQC \
	--fc_group_features gene_id \
	-r 1.4.2 \
	-c /lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/tmp/star-one-sample/myriad.config \
	-with-tower \
	-bg \
	-resume \
	--outdir /lustre/scratch/scratch/ucfaeef/comparative-transcriptomics/Polybia-occidentalis/result/star-one-sample
